12pub const S_CET: u32 = 0x6a2;
13pub const PL0_SSP: u32 = 0x6a4;
14pub const ISST_ADDR: u32 = 0x6a8;

16pub const MODE_64BIT: usize = 1;
17pub const BUSY: usize = 1;

19pub static IS_CET_SUPPORTED: AtomicBool = AtomicBool::new(false);
20
21// Try to enable the CET feature in CR4 and set `IS_CET_SUPPORTED` if successful.
22pub fn determine_cet_support(platform: &dyn SvsmPlatform) {
23    if platform.determine_cet_support() {
24        IS_CET_SUPPORTED.store(true, Ordering::Relaxed);
25    }
26}
27
28pub fn determine_cet_support_from_cpuid() -> bool {
29    let cpuid = CpuidResult::get(7, 0);
30    (cpuid.ecx & 0x80) != 0
31}

33/// Returns whether shadow stacks are supported by the CPU and the kernel.
34#[inline(always)]
35pub fn is_cet_ss_supported() -> bool {
36    // In theory CPUs can have support for CET, but not CET_SS, but in practice
37    // no such CPUs exist. Treat CET being supported as CET_SS being supported.
38    IS_CET_SUPPORTED.load(Ordering::Relaxed)
39}

41/// Enable shadow stacks.
42///
43/// This code is placed in a macro instead of a function so that we don't have
44/// to set up the shadow stack to return from this code.
45#[macro_export]
46macro_rules! enable_shadow_stacks {
47    ($bsp_percpu:ident) => {{
48        use core::arch::asm;
49
50        let token_addr = $bsp_percpu.get_top_of_shadow_stack();
51
52        unsafe {
53            asm!(
54                // Enable shadow stacks.
55                "wrmsr",
56                // Write a shadow stack restore token onto the stack.
57                "wrssq [{token_addr}], {token_val}",
58                // Load the shadow stack.
59                "rstorssp [{token_addr}]",
60                in("ecx") S_CET,
61                in("edx") 0,
62                in("eax") SCetFlags::SH_STK_EN.bits() | SCetFlags::WR_SHSTK_EN.bits(),
63                token_addr = in(reg) token_addr.bits(),
64                token_val = in(reg) token_addr.bits() + 8 + MODE_64BIT,
65                options(nostack, readonly),
66            );
67        }
68    }};
69}

71pub fn read_s_cet() -> SCetFlags {
72    SCetFlags::from_bits_retain(read_msr(S_CET))
73}
74
75bitflags! {
76    pub struct SCetFlags: u64 {
77        const SH_STK_EN = 1 << 0; // Enables the shadow stacks
78        const WR_SHSTK_EN = 1 << 1; // Enables the WRSS instruction
79    }
