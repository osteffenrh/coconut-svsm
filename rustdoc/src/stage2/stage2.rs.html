54fn setup_stage2_allocator(heap_start: u64, heap_end: u64) {
55    let vstart = VirtAddr::from(heap_start);
56    let vend = VirtAddr::from(heap_end);
57    let pstart = PhysAddr::from(vstart.bits()); // Identity mapping
58    let nr_pages = (vend - vstart) / PAGE_SIZE;
59
60    root_mem_init(pstart, vstart, nr_pages);
61}

63fn init_percpu(platform: &mut dyn SvsmPlatform) -> Result<(), SvsmError> {
64    // SAFETY: this is the first CPU, so there can be no other dependencies
65    // on multi-threaded access to the per-cpu areas.
66    let percpu_shared = unsafe { PERCPU_AREAS.create_new(0) };
67    let bsp_percpu = PerCpu::alloc(percpu_shared)?;
68    // SAFETY: pgtable is properly aligned and is never freed within the
69    // lifetime of stage2. We go through a raw pointer to promote it to a
70    // static mut. Only the BSP is able to get a reference to it so no
71    // aliasing can occur.
72    let init_pgtable = unsafe { (&raw mut pgtable).as_mut().unwrap() };
73    bsp_percpu.set_pgtable(init_pgtable);
74    bsp_percpu.map_self_stage2()?;
75    platform.setup_guest_host_comm(bsp_percpu, true);
76    Ok(())
77}

79/// Release all resources in the `PerCpu` instance associated with the current
80/// CPU.
81///
82/// # Safety
83///
84/// The caller must ensure that the `PerCpu` is never used again.
85unsafe fn shutdown_percpu() {
86    let ptr = SVSM_PERCPU_BASE.as_mut_ptr::<PerCpu>();
87    // SAFETY: ptr is properly aligned but the caller must ensure the PerCpu
88    // structure is valid and not aliased.
89    unsafe {
90        core::ptr::drop_in_place(ptr);
91    }
92    // SAFETY: pgtable is properly aligned and is never freed within the
93    // lifetime of stage2. We go through a raw pointer to promote it to a
94    // static mut. Only the BSP is able to get a reference to it so no
95    // aliasing can occur.
96    let init_pgtable = unsafe { (&raw mut pgtable).as_mut().unwrap() };
97    init_pgtable.unmap_4k(SVSM_PERCPU_BASE);
98    flush_tlb_percpu();
99}

101// SAFETY: the caller must guarantee that the IDT specified here will remain
102// in scope until a new IDT is loaded.
103unsafe fn setup_env(
104    config: &SvsmConfig<'_>,
105    platform: &mut dyn SvsmPlatform,
106    launch_info: &Stage2LaunchInfo,
107    idt: &mut IDT<'_>,
108) {
109    GLOBAL_GDT.load_selectors();
110    // SAFETY: the caller guarantees that the lifetime of this IDT is suitable.
111    unsafe {
112        early_idt_init_no_ghcb(idt);
113    }
114
115    let debug_serial_port = config.debug_serial_port();
116    install_console_logger("Stage2").expect("Console logger already initialized");
117    platform
118        .env_setup(debug_serial_port, launch_info.vtom.try_into().unwrap())
119        .expect("Early environment setup failed");
120
121    let kernel_mapping = FixedAddressMappingRange::new(
122        VirtAddr::from(u64::from(STAGE2_START)),
123        VirtAddr::from(u64::from(launch_info.stage2_end)),
124        PhysAddr::from(u64::from(STAGE2_START)),
125    );
126
127    // SAFETY: the CPUID page address specified in the launch info was mapped
128    // by the loader, which promises to supply a correctly formed CPUID page
129    // at that address.
130    let cpuid_page = unsafe { &*(launch_info.cpuid_page as *const SnpCpuidTable) };
131    register_cpuid_table(cpuid_page);
132    paging_init(platform, true).expect("Failed to initialize early paging");
133
134    // Use the low 640 KB of memory as the heap.
135    let lowmem_region =
136        MemoryRegion::from_addresses(VirtAddr::from(0u64), VirtAddr::from(u64::from(LOWMEM_END)));
137    let heap_mapping = FixedAddressMappingRange::new(
138        lowmem_region.start(),
139        lowmem_region.end(),
140        PhysAddr::from(0u64),
141    );
142    init_kernel_mapping_info(kernel_mapping, Some(heap_mapping));
143
144    // Now that the heap virtual-to-physical mapping has been established,
145    // validate the first 640 KB of memory so it can be used if necessary.
146    // SAFETY: the low memory region is known not to overlap any memory in use.
147    unsafe {
148        platform
149            .validate_virtual_page_range(lowmem_region, PageValidateOp::Validate)
150            .expect("failed to validate low 640 KB");
151    }
152
153    // SAFETY: ap_flag is an extern static and this is the only place where we
154    // get a reference to it.
155    unsafe {
156        // Allow APs to proceed as the environment is now ready.
157        //
158        // Although APs use non-atomic loads in the ap_wait_for_env spin loop,
159        // the language and architectural guarantees of this atomic store (e.g.
160        // the compiler cannot move the previous stores past this atomic
161        // store-release, and x86 is a strongly-ordered system) make setting
162        // this flag more deterministic.
163        ap_flag.store(1, Ordering::Release);
164    }
165
166    // Configure the heap.
167    setup_stage2_allocator(STAGE2_HEAP_START.into(), STAGE2_HEAP_END.into());
168
169    init_percpu(platform).expect("Failed to initialize per-cpu area");
170
171    // Init IDT again with handlers requiring GHCB (eg. #VC handler)
172    // SAFETY: the caller guarantees that the lifetime of this IDT is suitable.
173    unsafe {
174        early_idt_init(idt);
175    }
176
177    // Complete initializtion of the platform.  After that point, the console
178    // will be fully working and any unsupported configuration can be properly
179    // reported.
180    platform
181        .env_setup_late(debug_serial_port)
182        .expect("Late environment setup failed");
183
184    dump_cpuid_table();
185}

187/// Map and validate the specified virtual memory region at the given physical
188/// address.
189/// # Safety
190/// The caller is required to ensure the safety of the virtual address range.
191unsafe fn map_and_validate(
192    platform: &dyn SvsmPlatform,
193    config: &SvsmConfig<'_>,
194    vregion: MemoryRegion<VirtAddr>,
195    paddr: PhysAddr,
196) -> Result<(), SvsmError> {
197    let flags = PTEntryFlags::PRESENT
198        | PTEntryFlags::WRITABLE
199        | PTEntryFlags::ACCESSED
200        | PTEntryFlags::DIRTY;
201
202    let mut pgtbl = this_cpu().get_pgtable();
203    pgtbl.map_region(vregion, paddr, flags)?;
204
205    if config.page_state_change_required() {
206        platform.page_state_change(
207            MemoryRegion::new(paddr, vregion.len()),
208            PageSize::Huge,
209            PageStateChangeOp::Private,
210        )?;
211    }
212    // SAFETY: the caller has ensured the safety of the virtual address range.
213    unsafe {
214        platform.validate_virtual_page_range(vregion, PageValidateOp::Validate)?;
215    }
216    valid_bitmap_set_valid_range(paddr, paddr + vregion.len());
217    Ok(())
218}

220#[inline]
221fn check_launch_info(launch_info: &KernelLaunchInfo) {
222    let offset: u64 = launch_info.heap_area_virt_start - launch_info.heap_area_phys_start;
223    let align: u64 = PAGE_SIZE_2M.try_into().unwrap();
224
225    assert!(is_aligned(offset, align));
226}

228fn get_svsm_config(
229    launch_info: &Stage2LaunchInfo,
230    platform: &dyn SvsmPlatform,
231) -> Result<SvsmConfig<'static>, SvsmError> {
232    let igvm_params = if launch_info.igvm_params == 0 {
233        None
234    } else {
235        Some(IgvmParams::new(VirtAddr::from(

243/// Loads a single ELF segment and returns its virtual memory region.
244/// # Safety
245/// The caller is required to supply an appropriate virtual address for this
246/// ELF segment.
247unsafe fn load_elf_segment(
248    segment: elf::Elf64ImageLoadSegment<'_>,
249    paddr: PhysAddr,
250    platform: &dyn SvsmPlatform,
251    config: &SvsmConfig<'_>,
252) -> Result<MemoryRegion<VirtAddr>, SvsmError> {
253    // Find the segment's bounds
254    let segment_start = VirtAddr::from(segment.vaddr_range.vaddr_begin);
255    let segment_end = VirtAddr::from(segment.vaddr_range.vaddr_end).page_align_up();
256    let segment_len = segment_end - segment_start;
257    let segment_region = MemoryRegion::new(segment_start, segment_len);
258
259    // All ELF segments should be aligned to the page size. If not, there's
260    // the risk of pvalidating a page twice, bail out if so. Note that the
261    // ELF reading code had already verified that the individual segments,
262    // with bounds specified as in the ELF file, are non-overlapping.
263    if !segment_start.is_page_aligned() {
264        return Err(SvsmError::Elf(ElfError::UnalignedSegmentAddress));
265    }
266
267    // Map and validate the segment at the next contiguous physical address
268    // SAFETY: the caller has verified the safety of this virtual address
269    // region.
270    unsafe {
271        map_and_validate(platform, config, segment_region, paddr)?;
272    }
273
274    // Copy the segment contents and pad with zeroes
275    // SAFETY: the caller guarantees the correctness of the ELF segment's
276    // virtual address range.
277    let segment_buf =
278        unsafe { slice::from_raw_parts_mut(segment_start.as_mut_ptr::<u8>(), segment_len) };
279    let contents_len = segment.file_contents.len();
280    segment_buf[..contents_len].copy_from_slice(segment.file_contents);
281    segment_buf[contents_len..].fill(0);
282
283    Ok(segment_region)
284}

286/// Loads the kernel ELF and returns the virtual memory region where it
287/// resides, as well as its entry point. Updates the used physical memory
288/// region accordingly.
289fn load_kernel_elf(
290    launch_info: &Stage2LaunchInfo,
291    loaded_phys: &mut MemoryRegion<PhysAddr>,
292    platform: &dyn SvsmPlatform,
293    config: &SvsmConfig<'_>,
294) -> Result<(VirtAddr, MemoryRegion<VirtAddr>), SvsmError> {
295    // Find the bounds of the kernel ELF and load it into the ELF parser
296    let elf_start = PhysAddr::from(launch_info.kernel_elf_start as u64);
297    let elf_end = PhysAddr::from(launch_info.kernel_elf_end as u64);
298    let elf_len = elf_end - elf_start;
299    // SAFETY: the base address of the ELF image was selected by the loader and
300    // is known not to conflict with any other virtual address mappings.
301    let bytes = unsafe { slice::from_raw_parts(elf_start.bits() as *const u8, elf_len) };
302    let elf = elf::Elf64File::read(bytes)?;
303
304    let vaddr_alloc_info = elf.image_load_vaddr_alloc_info();
305    let vaddr_alloc_base = vaddr_alloc_info.range.vaddr_begin;
306
307    // Map, validate and populate the SVSM kernel ELF's PT_LOAD segments. The
308    // segments' virtual address range might not necessarily be contiguous,
309    // track their total extent along the way. Physical memory is successively
310    // being taken from the physical memory region, the remaining space will be
311    // available as heap space for the SVSM kernel. Remember the end of all
312    // physical memory occupied by the loaded ELF image.
313    let mut load_virt_start = None;
314    let mut load_virt_end = VirtAddr::null();
315    for segment in elf.image_load_segment_iter(vaddr_alloc_base) {
316        // SAFETY: the virtual address is calculated based on the base address
317        // of the image and the previously loaded segments, so it is correct
318        // for use.
319        let region = unsafe { load_elf_segment(segment, loaded_phys.end(), platform, config)? };
320        // Remember the mapping range's lower and upper bounds to pass it on
321        // the kernel later. Note that the segments are being iterated over
322        // here in increasing load order.
323        if load_virt_start.is_none() {
324            load_virt_start = Some(region.start());
325        }
326        load_virt_end = region.end();
327
328        // Update to the next contiguous physical address
329        *loaded_phys = loaded_phys.expand(region.len());
330    }
331
332    let Some(load_virt_start) = load_virt_start else {
333        log::error!("No loadable segment found in kernel ELF");
334        return Err(SvsmError::Mem);
335    };
336
337    // Apply relocations, if any
338    if let Some(dyn_relocs) =
339        elf.apply_dyn_relas(elf::Elf64X86RelocProcessor::new(), vaddr_alloc_base)?
340    {
341        for reloc in dyn_relocs {
342            let Some(reloc) = reloc? else {
343                continue;
344            };
345            // TODO: ensure that the ELF package rejects illegal relocations
346            // the point outside the image.
347            // SAFETY: the relocation address is known to be correct.
348            let dst = unsafe { slice::from_raw_parts_mut(reloc.dst as *mut u8, reloc.value_len) };
349            let src = &reloc.value[..reloc.value_len];
350            dst.copy_from_slice(src)
351        }
352    }
353
354    let entry = VirtAddr::from(elf.get_entry(vaddr_alloc_base));
355    let region = MemoryRegion::from_addresses(load_virt_start, load_virt_end);
356    Ok((entry, region))
357}

359/// Loads the IGVM params at the next contiguous location from the loaded
360/// kernel image. Returns the virtual and physical memory regions hosting the
361/// loaded data.
362/// # Safety
363/// Ther caller is required to specify the correct virtual address for the
364/// kernel virtual region.
365unsafe fn load_igvm_params(
366    launch_info: &Stage2LaunchInfo,
367    params: &IgvmParams<'_>,
368    loaded_kernel_vregion: &MemoryRegion<VirtAddr>,
369    loaded_kernel_pregion: &MemoryRegion<PhysAddr>,
370    platform: &dyn SvsmPlatform,
371    config: &SvsmConfig<'_>,
372) -> Result<(MemoryRegion<VirtAddr>, MemoryRegion<PhysAddr>), SvsmError> {
373    // Map and validate destination region
374    let igvm_vregion = MemoryRegion::new(loaded_kernel_vregion.end(), params.size());
375    let igvm_pregion = MemoryRegion::new(loaded_kernel_pregion.end(), params.size());
376    // SAFETY: the virtual address region was computed not to overlap the
377    // kernel image.
378    unsafe {
379        map_and_validate(platform, config, igvm_vregion, igvm_pregion.start())?;
380    }
381
382    // Copy the contents over
383    let src_addr = VirtAddr::from(launch_info.igvm_params as u64);
384    // SAFETY: the source address specified in the launch info was mapped
385    // by the loader, which promises to supply a correctly formed IGRM
386    // parameter block
387    let igvm_src = unsafe { slice::from_raw_parts(src_addr.as_ptr::<u8>(), igvm_vregion.len()) };
388    // SAFETY: the destination address is calculated to follow the kernel ELF
389    // image, which the caller is required to calculate correctly.
390    let igvm_dst = unsafe {
391        slice::from_raw_parts_mut(igvm_vregion.start().as_mut_ptr::<u8>(), igvm_vregion.len())
392    };
393    igvm_dst.copy_from_slice(igvm_src);
394
395    Ok((igvm_vregion, igvm_pregion))
396}

398/// Maps any remaining memory between the end of the kernel image and the end
399/// of the allocated kernel memory region as heap space. Exclude any memory
400/// reserved by the configuration.
401///
402/// # Panics
403///
404/// Panics if the allocated kernel region (`kernel_region`) is not sufficient
405/// to host the loaded kernel region (`loaded_kernel_pregion`) plus memory
406/// reserved for configuration.
407fn prepare_heap(
408    kernel_region: MemoryRegion<PhysAddr>,
409    loaded_kernel_pregion: MemoryRegion<PhysAddr>,
410    loaded_kernel_vregion: MemoryRegion<VirtAddr>,
411    platform: &dyn SvsmPlatform,
412    config: &SvsmConfig<'_>,
413) -> Result<(MemoryRegion<VirtAddr>, MemoryRegion<PhysAddr>), SvsmError> {
414    // Heap starts after kernel
415    let heap_pstart = loaded_kernel_pregion.end();
416    let heap_vstart = loaded_kernel_vregion.end();
417
418    // Compute size, excluding any memory reserved by the configuration.
419    let heap_size = kernel_region
420        .end()
421        .checked_sub(heap_pstart.into())
422        .and_then(|r| r.checked_sub(config.reserved_kernel_area_size()))
423        .expect("Insufficient physical space for kernel image")
424        .into();
425    let heap_pregion = MemoryRegion::new(heap_pstart, heap_size);
426    let heap_vregion = MemoryRegion::new(heap_vstart, heap_size);
427
428    // SAFETY: the virtual address range was computed so it is within the valid
429    // region and does not collide with the kernel.
430    unsafe {
431        map_and_validate(platform, config, heap_vregion, heap_pregion.start())?;
432    }
433
434    Ok((heap_vregion, heap_pregion))
435}

437#[no_mangle]
438pub extern "C" fn stage2_main(launch_info: &Stage2LaunchInfo) -> ! {
439    let platform_type = SvsmPlatformType::from(launch_info.platform_type);
440
441    init_platform_type(platform_type);
442    let mut platform_cell = SvsmPlatformCell::new(true);
443    let platform = platform_cell.platform_mut();
444
445    let config = get_svsm_config(launch_info, platform).expect("Failed to get SVSM configuration");
446
447    // Set up space for an early IDT.  This will remain in scope as long as
448    // stage2 is in memory.
449    let mut early_idt = [IdtEntry::no_handler(); EARLY_IDT_ENTRIES];
450    let mut idt = IDT::new(&mut early_idt);
451
452    // SAFETY: the IDT here will remain in scope until the full IDT is
453    // initialized later, and thus can safely be used as the early IDT.
454    unsafe {
455        setup_env(&config, platform, launch_info, &mut idt);
456    }
457
458    // Get the available physical memory region for the kernel
459    let kernel_region = config
460        .find_kernel_region()
461        .expect("Failed to find memory region for SVSM kernel");
462
463    log::info!("SVSM memory region: {kernel_region:#018x}");
464
465    init_valid_bitmap_alloc(kernel_region).expect("Failed to allocate valid-bitmap");
466
467    // The physical memory region we've loaded so far
468    let mut loaded_kernel_pregion = MemoryRegion::new(kernel_region.start(), 0);
469
470    // Load first the kernel ELF and update the loaded physical region
471    let (kernel_entry, mut loaded_kernel_vregion) =
472        load_kernel_elf(launch_info, &mut loaded_kernel_pregion, platform, &config)
473            .expect("Failed to load kernel ELF");
474
475    // Load the IGVM params, if present. Update loaded region accordingly.
476    let (igvm_vregion, igvm_pregion) = if let Some(igvm_params) = config.get_igvm_params() {
477        // SAFETY: The loaded kernel region was correctly calculated above and
478        // is sized appropriately to include a copy of the IGVM parameters.
479        let (igvm_vregion, igvm_pregion) = unsafe {
480            load_igvm_params(
481                launch_info,
482                igvm_params,
483                &loaded_kernel_vregion,
484                &loaded_kernel_pregion,
485                platform,
486                &config,
487            )
488        }
489        .expect("Failed to load IGVM params");
490
491        // Update the loaded kernel region
492        loaded_kernel_pregion = loaded_kernel_pregion.expand(igvm_vregion.len());
493        loaded_kernel_vregion = loaded_kernel_vregion.expand(igvm_pregion.len());
494        (igvm_vregion, igvm_pregion)
495    } else {
496        (
497            MemoryRegion::new(VirtAddr::null(), 0),
498            MemoryRegion::new(PhysAddr::null(), 0),
499        )
500    };
501
502    // Use remaining space after kernel image as heap space.
503    let (heap_vregion, heap_pregion) = prepare_heap(
504        kernel_region,
505        loaded_kernel_pregion,
506        loaded_kernel_vregion,
507        platform,
508        &config,
509    )
510    .expect("Failed to map and validate heap");
511
512    // Determine whether use of interrupts n the SVSM should be suppressed.
513    // This is required when running SNP under KVM/QEMU.
514    let suppress_svsm_interrupts = match platform_type {
515        SvsmPlatformType::Snp => config.is_qemu(),
516        _ => false,
517    };
518
519    // Build the handover information describing the memory layout and hand
520    // control to the SVSM kernel.
521    let launch_info = KernelLaunchInfo {
522        kernel_region_phys_start: u64::from(kernel_region.start()),
523        kernel_region_phys_end: u64::from(kernel_region.end()),
524        heap_area_phys_start: u64::from(heap_pregion.start()),
525        heap_area_virt_start: u64::from(heap_vregion.start()),
526        heap_area_size: heap_vregion.len() as u64,
527        kernel_region_virt_start: u64::from(loaded_kernel_vregion.start()),
528        kernel_elf_stage2_virt_start: u64::from(launch_info.kernel_elf_start),
529        kernel_elf_stage2_virt_end: u64::from(launch_info.kernel_elf_end),
530        kernel_fs_start: u64::from(launch_info.kernel_fs_start),
531        kernel_fs_end: u64::from(launch_info.kernel_fs_end),
532        stage2_start: 0x800000u64,
533        stage2_end: launch_info.stage2_end as u64,
534        cpuid_page: launch_info.cpuid_page as u64,
535        secrets_page: launch_info.secrets_page as u64,
536        stage2_igvm_params_phys_addr: u64::from(launch_info.igvm_params),
537        stage2_igvm_params_size: igvm_pregion.len() as u64,
538        igvm_params_phys_addr: u64::from(igvm_pregion.start()),
539        igvm_params_virt_addr: u64::from(igvm_vregion.start()),
540        vtom: launch_info.vtom,
541        debug_serial_port: config.debug_serial_port(),
542        use_alternate_injection: config.use_alternate_injection(),
543        suppress_svsm_interrupts,
544        platform_type,
545    };
546
547    check_launch_info(&launch_info);
548
549    let mem_info = memory_info();
550    print_memory_info(&mem_info);
551
552    log::info!(
553        "  kernel_region_phys_start = {:#018x}",
554        kernel_region.start()
555    );
556    log::info!("  kernel_region_phys_end   = {:#018x}", kernel_region.end());
557    log::info!(
558        "  kernel_virtual_base      = {:#018x}",
559        loaded_kernel_vregion.start()
560    );
561
562    let valid_bitmap = valid_bitmap_addr();
563
564    log::info!("Starting SVSM kernel...");
565
566    // Shut down the GHCB
567    // SAFETY: the addreses used to invoke the kernel have been calculated
568    // correctly for use in the assembly trampoline.
569    unsafe {
570        shutdown_percpu();
571
572        asm!("jmp *%rax",
573             in("rax") u64::from(kernel_entry),
574             in("rdi") &launch_info,
575             in("rsi") valid_bitmap.bits(),
576             options(att_syntax))
577    };
578
579    unreachable!("Road ends here!");
580}
